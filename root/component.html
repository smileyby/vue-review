# 组件库开发流程

// 引入组件库方式
import ElementUI from 'element-ui'
Vue.use(ElementUI);

// 开发自己的组件库
// 路径：./packages/index.js
import button from './button.vue';
button.install = function(_vue) { // Vue.use 会调用 install方法
    _vue.component('my-button', button);
}
export default button.install;
